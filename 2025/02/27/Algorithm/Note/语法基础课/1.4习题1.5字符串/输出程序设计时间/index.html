#include <ctime>

using namespace std;

int main(){
    int start_time = clock();//clock会返回这个程序的开始时刻 这个函数在ctime里面
    int end_time = clock() - start_time;
    string a;
    getline(cin, a);
    printf("%s\n%d", a, end_time);
    
    return 0;
}
